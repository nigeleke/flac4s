43 object MetadataBlocks:
44   //  <1>   Last-metadata-block flag: '1' if this block is the last metadata block before the audio blocks, '0' otherwise.
45   //  <7>	  BLOCK_TYPE
46   //        0 : STREAMINFO
47   //        1 : PADDING
48   //        2 : APPLICATION
49   //        3 : SEEKTABLE
50   //        4 : VORBIS_COMMENT
51   //        5 : CUESHEET
52   //        6 : PICTURE
53   //        7-126 : reserved
54   //        127 : invalid, to avoid confusion with a frame sync code
55   //  <24>	Length (in bytes) of metadata to follow (does not include the size of the METADATA_BLOCK_HEADER)
56   //
57   // List[MetadataBlock] is NonEmpty as StreamInfo is a mandatory first block.
58   //
59   val codec: Codec[MetadataBlocks] =
60     MetadataBlock.codec.consume(block =>
61       if block.header.isLast
62       then provide(List(block))
63       else codec.xmap(blocks => block +: blocks, blocks => blocks.tail)
64     )(blocks => blocks.head)
