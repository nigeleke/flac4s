42 object SeekPoint:
43 
44   //  <64>	Sample number of first sample in the target frame, or 0xFFFFFFFFFFFFFFFF for a placeholder point.
45   //  <64>	Offset (in bytes) from the first byte of the first frame header to the first byte of the target frame's header.
46   //  <16>	Number of samples in the target frame.
47   //
48   private val sampleNumberCodec      = logField("sampleNumber")(int64)

50   private val samplesCountCodec      = logField("samplesCount")(int16)
51

53     sampleNumberCodec :: frameHeaderOffsetCodec :: samplesCountCodec
54   }.dropUnits.as[SeekPoint]
