44 final case class MetadataBlock(
45     header: MetadataBlock.Header,
46     data: MetadataBlock.Data
47 )
48 
49 object MetadataBlock:
50 
51   enum Type:
52     case StreamInfo, Padding, Application, SeekTable, VorbisComment, Cuesheet, Picture
53 
54   final case class Header(isLast: Boolean, blockType: Type, length: Int)
55   trait Data
56 
57   private val isLastCodec    = logField("isLast")(bool)
58   private val blockTypeCodec = logField("blockType")(uint(7).xmap(Type.fromOrdinal, _.ordinal))
59   private val lengthCodec    = logField("length")(uint24)
60 
61   private val headerCodec = {
62     isLastCodec :: blockTypeCodec :: lengthCodec
63   }.as[Header]
64 
65   //  <1>   Last-metadata-block flag: '1' if this block is the last metadata block before the audio blocks, '0' otherwise.
66   //  <7>	  BLOCK_TYPE
67   //        0 : STREAMINFO
68   //        1 : PADDING
69   //        2 : APPLICATION
70   //        3 : SEEKTABLE
71   //        4 : VORBIS_COMMENT
72   //        5 : CUESHEET
73   //        6 : PICTURE
74   //        7-126 : reserved
75   //        127 : invalid, to avoid confusion with a frame sync code
76   //  <24>	Length (in bytes) of metadata to follow (does not include the size of the METADATA_BLOCK_HEADER)
77   private def dataCodec(header: Header): Codec[Data] =
78     given Header = header
79     discriminated[Data]
80       .by(provide(header.blockType))
81       .typecase(Type.Application, Application.codec)
82       .typecase(Type.Cuesheet, Cuesheet.codec)
83       .typecase(Type.Padding, Padding.codec)
84       .typecase(Type.Picture, Picture.codec)
85       .typecase(Type.SeekTable, SeekTable.codec)
86       .typecase(Type.StreamInfo, StreamInfo.codec)
87       .typecase(Type.VorbisComment, VorbisComment.codec)
88 
89   val codec: Codec[MetadataBlock] = logMetadata("metadataBlock") {
90     headerCodec flatZip dataCodec
91   }.dropUnits.as[MetadataBlock]
