41 final case class VorbisComment(vendor: String, comments: Vector[String]) extends MetadataBlock.Data
42 
43 object VorbisComment:
44 
45   //  The comment header is decoded as follows:
46   //
47   //  1) [vendor_length] = read an unsigned integer of 32 bits
48   //  2) [vendor_string] = read a UTF-8 vector as [vendor_length] octets
49   //  3) [user_comment_list_length] = read an unsigned integer of 32 bits
50   //  4) iterate [user_comment_list_length] times {
51   //      5) [length] = read an unsigned integer of 32 bits
52   //      6) this iteration's user comment = read a UTF-8 vector as [length] octets
53   //  7) [framing_bit] = read a single bit as boolean
54   //  8) -- FlacApp spec states "without the framing bit" if ( [framing_bit] unset or end of packet ) then ERROR
55   //  9) done.
56   //
57 
58   // TODO: Flac spec states should be unsigned; scodec documentation states signed
59   private val vendorCodec = logField("vendor")(utf8_32L)
60 
61   // TODO: Flac spec states should be unsigned; scodec documentation states signed
62   private val commentCodec = logField("comment")(utf8_32L)
63 
64   // TODO: Flac spec states unsigned long. Can't hold this in Vector.
65   private val commentsCodec = vectorOfN(int32L, commentCodec)
66 
67   val codec: Codec[VorbisComment] = {
68     (vendorCodec :: commentsCodec)
69   }.dropUnits.as[VorbisComment]
