39 object Logger:
40 
41   lazy val config             = ConfigFactory.load()
42   lazy val configuredlogLevel = LogLevel.valueOf(config.getString("flac4s.logging.logLevel"))
43 
44   enum LogLevel:
45     case None, Metadata, Field, Audio
46 
47   private def log[A](name: String, logLevel: LogLevel)(codec: Codec[A]) =
48     val logThis = configuredlogLevel.ordinal >= logLevel.ordinal
49     if logThis
50     then logToStdOut((name | codec), name)
51     else (name | codec)
52 
53   def logField[A](name: String)(codec: Codec[A])    = log(name, LogLevel.Field)(codec)
54   def logMetadata[A](name: String)(codec: Codec[A]) = log(name, LogLevel.Metadata)(codec)
55   def logAudio[A](name: String)(codec: Codec[A])    = log(name, LogLevel.Audio)(codec)
